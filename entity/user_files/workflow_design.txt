```plaintext
        "buildType": {
            "id": "KubernetesPipeline_CyodaSaasUserEnv"
        },

            ]
        }
    }

    // Call TeamCity API to trigger the deployment
    deployment_response = callApi("POST", "https://teamcity.cyoda.org/app/rest/buildQueue", deployment_params)

    // Check for a successful deployment trigger
    if deployment_response.status == 200:
        // Log successful deployment
        log("Deployment triggered successfully for user app with ID: " + deployment_response.build_id)
        // Update entity status to reflect the deployment process
        update_deployment_status(current_entity.id, "deployed", deployment_response.build_id)
    else:
        // Log failure for the deployment trigger
        log("Failed to trigger deployment for user app with ID: " + current_entity.id + " Response: " + deployment_response.status)
        update_deployment_status(current_entity.id, "deployment_failed")
```
